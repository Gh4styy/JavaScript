                <script language="JavaScript">
                    do {
                        var liczba1 = prompt("Podaj bok a=", "");
                        var a_for = parseFloat(liczba1);
                    } while (a_for <= 0);

                    do {
                        var liczba2 = prompt("Podaj bok b=", "");
                        var b_for = parseFloat(liczba2);
                    } while (b_for <= 0);

                    do {
                        var liczba3 = prompt("Podaj bok c=", "");
                        var c_for = parseFloat(liczba3);
                    } while (c_for <= 0);

                    if (a_for + b_for > c_for && a_for + c_for > b_for && b_for + c_for > a_for) {
                        document.write("Boki tworzą trójkąt" + "<br>");
                    } else {
                        document.write("Boki nie tworzą trójkąta" + "<br>");
                    }
